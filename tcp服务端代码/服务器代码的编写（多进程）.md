### 服务器代码的编写（多进程）

问题：如果信号注册之前，子进程退出，无法捕捉信号；如果信号提前注册，子进程会复制父进程流程

解决方法：注册之前，把信号给屏蔽住；不屏蔽住的画，父进程会忽略信号。



~~~c++
#include <stdio.h>
#include <sys/socket.h>
#include <unistd.h>
#include <signal.h>
#include <sys/wait.h>
#include "wrap.h"
void free_process(int sig)     
{
	pid_t pid;
	while(1)
	{
		pid = waitpid(-1,NULL,WNOHANG);
		if(pid <=0 )//小于0 子进程全部退出了 =0没有进程没有退出
		{
			break;
		}
		else
		{
			printf("child pid =%d\n",pid);
		}
	}



}
int main(int argc, char *argv[])
{
	sigset_t set;    //设置阻塞集
	sigemptyset(&set);//清空阻塞集
	sigaddset(&set,SIGCHLD);//将SIGCHLD（子进程终止）信号添加到集合里
	sigprocmask(SIG_BLOCK,&set,NULL);//将信号添加到阻塞集合中
    //注：SIG_UNBLOCK:将参数set所指向的信号集中的信号从当前的信号掩码中移除
    //   SIG_BLOCK：set包含了我们希望添加到当前信号屏蔽字的信号
    
	//创建套接字,绑定
	int lfd = tcp4bind(8008,NULL);
	//监控
	Listen(lfd,128);
	//提取
	//回射
	struct sockaddr_in cliaddr;
	socklen_t len = sizeof(cliaddr);
	while(1)
	{
		char ip[16]="";
		//提取连接
		int cfd = Accept(lfd,(struct sockaddr *)&cliaddr,&len);
		printf("new client ip=%s port=%d\n",inet_ntop(AF_INET,&cliaddr.sin_addr.s_addr,ip,16),
				ntohs(cliaddr.sin_port));//打印客户端IP和端口号
		//fork创建子进程
		pid_t pid;
		pid = fork();
		if(pid < 0)
		{
			perror("");
			exit(0);
		}
		else if(pid == 0)//子进程
		{
			//关闭lfd
			close(lfd);
			while(1)
			{
			char buf[1024]="";

			int n = read(cfd,buf,sizeof(buf));
			if(n < 0)
			{
				perror("");
				close(cfd);
				exit(0);
			}
			else if(n == 0)//对方关闭
			{
				printf("client close\n");
				close(cfd);
				exit(0);
			
			}
			else
			{
				printf("%s\n",buf);
				write(cfd,buf,n);
			//	exit(0);	
			}
			}
		
		}
		else//父进程
		{
			close(cfd);
			//回收
			//注册信号回调
			struct sigaction act;
			act.sa_flags =0;
			act.sa_handler = free_process;
			sigemptyset(&act.sa_mask);
			sigaction(SIGCHLD,&act,NULL);
            //解开阻塞集合
			sigprocmask(SIG_UNBLOCK,&set,NULL);
		
		}
	}
	//关闭



	return 0;
}
~~~
